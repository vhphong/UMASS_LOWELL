I. Command lines:
./mcc hw4symasm.mc > prom_new.dat //translate microcode source file into a 32 bit microcode.dat
make                              // build masm program  
./hw4symasm_all <a4_div_test.asm> div.obj        //create object file of the a4_div_test file
./hw4symasm_all <a4_mult_test.asm> mult.obj	 //create object file of the a4_mult_test file
./hw4symasm_all <a4_rshift_test.asm> rshift.obj //create object file of the a4_rshift_test file
./mic1 prom_new.dat div.obj			//mic1 execution
./mic1 prom_new.dat mult.obj			//mic1 execution
./mic1 prom_new.dat rshift.obj			//mic1 execution
